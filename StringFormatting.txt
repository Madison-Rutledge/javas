public class StringFormatting
{
	public static void main(String[] args){
		System.out.println(String.format("The total price is $%.2f", 21.7));
		System.out.printf("%10d", 19);
		System.out.printf("%-9d", -37);

		System.out.println("\nMy Favourite Book is \"Insignia\".");
		System.out.println("Hello\nGoodbye");
	}
}
